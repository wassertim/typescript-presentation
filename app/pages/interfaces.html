    interface ICountry {
      title: string;
      getPopulation(): number;
    }
  

  
    var country: ICountry = {
      title: 'Germany',
      getPopulation: () => 80620000;
    }
  

  
    class Country implements ICountry {
      title = 'Russia',
      getPopulation(): number {
        return 143500000;
      }
    }

    var country = new Country();

Class Types

  
    interface ICity {
      title: string;
      population: number;
    }

    class City implements ICity {
      title: string;
      population: number;

      constructor(title: string, population: number) {
        this.title = title;
        this.population = population;
      }
    }

Extending Interfaces

      color: string;
    }

    interface Square extends Shape {
      sideLength: number;
    }

    var square = <Square>{};
    square.color = "blue";
    square.sideLength = 10;
